#### **해시값 생성 방법**
```javascript
// AI 도구로 생성 가능
const crypto = require('crypto');

function generateAnonymousId(phone, timestamp) {
  // 전화번호 + 타임스탬프로 해시 생성
  const data = phone + timestamp;
  return crypto.createHash('sha256').update(data).digest('hex');
}

// 예시
const anonymousId = generateAnonymousId('01012345678', Date.now());
// 결과: "a1b2c3d4e5f6..." (64자리 해시값)
```
