## 💡 **AI 도구로 암호화 구현**

### **1. 데이터베이스 암호화**

#### **Node.js 예시**
```javascript
// AI에게: "데이터베이스 암호화 코드 만들어줘"

const crypto = require('crypto');

// 암호화 함수
function encrypt(text, key) {
  const iv = crypto.randomBytes(16);
  const cipher = crypto.createCipheriv('aes-256-cbc', key, iv);
  let encrypted = cipher.update(text, 'utf8', 'hex');
  encrypted += cipher.final('hex');
  return iv.toString('hex') + ':' + encrypted;
}

// 복호화 함수
function decrypt(encryptedText, key) {
  const parts = encryptedText.split(':');
  const iv = Buffer.from(parts[0], 'hex');
  const encrypted = parts[1];
  const decipher = crypto.createDecipheriv('aes-256-cbc', key, iv);
  let decrypted = decipher.update(encrypted, 'hex', 'utf8');
  decrypted += decipher.final('utf8');
  return decrypted;
}
```

### **2. 개인정보 필드 암호화**

#### **사용 예시**
```javascript
// 저장 시
const encryptedPhone = encrypt(user.phone, encryptionKey);
db.saveUser({ ...user, phone: encryptedPhone });

// 조회 시
const user = db.getUser(id);
const decryptedPhone = decrypt(user.phone, encryptionKey);
```

### **3. 비밀번호 해싱**

#### **bcrypt 예시**
```javascript
// AI에게: "비밀번호 bcrypt로 해싱해줘"

const bcrypt = require('bcrypt');

// 저장 시
const hashedPassword = await bcrypt.hash(password, 10);
db.saveUser({ ...user, password: hashedPassword });

// 검증 시
const isValid = await bcrypt.compare(password, hashedPassword);
```
